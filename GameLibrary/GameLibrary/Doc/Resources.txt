// pass in a position and camera matrices
// returns true if the position is on screen
bool VertexIsOnScreen(Vector3 position3D, Matrix projMatrix, Matrix viewMatrix, Matrix worldMatrix)
{
  // project 3d position to 2d
  Vector2 position2D = GraphicsDeviceManager.GraphicsDevice.ViewPort.Project(position3D, projMatrix, viewMatrix, worldMatrix);

  // check x is on screen
  if(position2D.X < 0 || position2D.X > GraphicsDeviceManager.GraphicsDevice.PreferredBackBufferWidth) return false;

  // check y is on screen
  if(position2D.Y < 0 || position2D.Y > GraphicsDeviceManager.GraphicsDevice.PreferredBackBufferHeight) return false;

  return true;
}
